\mbox{\hyperlink{common_8h_ab33860705c91ca5736335f59d007845c}{MRB\+\_\+\+API}} void mrb\+\_\+vm\+\_\+ci\+\_\+env\+\_\+clear (\begin{DoxyParamCaption}\item[{\mbox{\hyperlink{structmrb__state}{mrb\+\_\+state}} \texorpdfstring{$\ast$}{*}}]{mrb}{, }\item[{\mbox{\hyperlink{structmrb__callinfo}{mrb\+\_\+callinfo}} \texorpdfstring{$\ast$}{*}}]{ci}{}\end{DoxyParamCaption})}

It can be used to isolate top-\/level scopes referenced by blocks generated by {\ttfamily mrb\+\_\+load\+\_\+string\+\_\+cxt()} or similar called before entering the mruby VM (e.\+g. from {\ttfamily main()}). In that case, the {\ttfamily ci} parameter should be {\ttfamily mrb-\/\texorpdfstring{$>$}{>}c-\/\texorpdfstring{$>$}{>}cibase}. \begin{DoxyVerb}#include <mruby.h>
#include <mruby/compile.h>
#include <mruby/proc.h>

int
main(int argc, char **argv)
{
  mrb_state *mrb;
  mrb_ccontext *cxt;
  mrb_value blk, ret;

  mrb = mrb_open();
  cxt = mrb_ccontext_new(mrb);
  blk = mrb_load_string_cxt(mrb, "x, y, z = 1, 2, 3; proc { [x, y, z] }", cxt);
  mrb_vm_ci_env_clear(mrb, mrb->c->cibase);
  mrb_load_string_cxt(mrb, "x, y, z = 4, 5, 6", cxt);
  ret = mrb_funcall(mrb, blk, "call", 0);
  mrb_p(mrb, ret);  // => [1, 2, 3]
                    // => [4, 5, 6] if `mrb_vm_ci_env_clear()` is commented out
  mrb_ccontext_free(mrb, cxt);
  mrb_close(mrb);

  return 0;
}
\end{DoxyVerb}


The top-\/level local variable names stored in {\ttfamily \doxylink{structmrb__ccontext}{mrb\+\_\+ccontext}} are retained. Use also {\ttfamily mrb\+\_\+ccontext\+\_\+cleanup\+\_\+local\+\_\+variables()} at the same time, if necessary.
