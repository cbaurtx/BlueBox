38/*
39 * If local peer is stream initiator:
40 * NGHTTP2_STREAM_OPENING : upon sending request HEADERS
41 * NGHTTP2_STREAM_OPENED : upon receiving response HEADERS
42 * NGHTTP2_STREAM_CLOSING : upon queuing RST_STREAM
43 *
44 * If remote peer is stream initiator:
45 * NGHTTP2_STREAM_OPENING : upon receiving request HEADERS
46 * NGHTTP2_STREAM_OPENED : upon sending response HEADERS
47 * NGHTTP2_STREAM_CLOSING : upon queuing RST_STREAM
48 */
49typedef enum {
50 /* Initial state */
51 NGHTTP2_STREAM_INITIAL,
52 /* For stream initiator: request HEADERS has been sent, but response
53 HEADERS has not been received yet. For receiver: request HEADERS
54 has been received, but it does not send response HEADERS yet. */
55 NGHTTP2_STREAM_OPENING,
56 /* For stream initiator: response HEADERS is received. For receiver:
57 response HEADERS is sent. */
58 NGHTTP2_STREAM_OPENED,
59 /* RST_STREAM is received, but somehow we need to keep stream in
60 memory. */
61 NGHTTP2_STREAM_CLOSING,
62 /* PUSH_PROMISE is received or sent */
63 NGHTTP2_STREAM_RESERVED,
64 /* Stream is created in this state if it is used as anchor in
65 dependency tree. */
66 NGHTTP2_STREAM_IDLE
67} nghttp2_stream_state;

69typedef enum {
70 NGHTTP2_SHUT_NONE = 0,
71 /* Indicates further receptions will be disallowed. */
72 NGHTTP2_SHUT_RD = 0x01,
73 /* Indicates further transmissions will be disallowed. */
74 NGHTTP2_SHUT_WR = 0x02,
75 /* Indicates both further receptions and transmissions will be
76 disallowed. */
77 NGHTTP2_SHUT_RDWR = NGHTTP2_SHUT_RD | NGHTTP2_SHUT_WR
78} nghttp2_shut_flag;

80typedef enum {
81 NGHTTP2_STREAM_FLAG_NONE = 0,
82 /* Indicates that this stream is pushed stream and not opened
83 yet. */
84 NGHTTP2_STREAM_FLAG_PUSH = 0x01,
85 /* Indicates that this stream was closed */
86 NGHTTP2_STREAM_FLAG_CLOSED = 0x02,
87 /* Indicates the item is deferred due to flow control. */
88 NGHTTP2_STREAM_FLAG_DEFERRED_FLOW_CONTROL = 0x04,
89 /* Indicates the item is deferred by user callback */
90 NGHTTP2_STREAM_FLAG_DEFERRED_USER = 0x08,
91 /* bitwise OR of NGHTTP2_STREAM_FLAG_DEFERRED_FLOW_CONTROL and
92 NGHTTP2_STREAM_FLAG_DEFERRED_USER. */
93 NGHTTP2_STREAM_FLAG_DEFERRED_ALL = 0x0c,
94 /* Ignore client RFC 9218 priority signal. */
95 NGHTTP2_STREAM_FLAG_IGNORE_CLIENT_PRIORITIES = 0x20,
96 /* Indicates that RFC 9113 leading and trailing white spaces
97 validation against a field value is not performed. */
98 NGHTTP2_STREAM_FLAG_NO_RFC9113_LEADING_AND_TRAILING_WS_VALIDATION = 0x40,
99} nghttp2_stream_flag;

102typedef enum {

145struct nghttp2_stream {
146 nghttp2_stream_state state;
147 nghttp2_pq_entry pq_entry;
148 /* Content-Length of request/response body. -1 if unknown. */
149 int64_t content_length;
150 /* Received body so far */
151 int64_t recv_content_length;
152 /* Next scheduled time to sent item */
153 uint64_t cycle;
154 /* Secondary key for prioritization to break a tie for cycle. This
155 value is monotonically increased for single parent stream. */
156 uint64_t seq;
157 nghttp2_stream *closed_next;
158 /* The arbitrary data provided by user for this stream. */
159 void *stream_user_data;
160 /* Item to send */
161 nghttp2_outbound_item *item;
162 /* Last written length of frame payload */
163 size_t last_writelen;
164 /* stream ID */
165 int32_t stream_id;
166 /* Current remote window size. This value is computed against the
167 current initial window size of remote endpoint. */
168 int32_t remote_window_size;
169 /* Keep track of the number of bytes received without
170 WINDOW_UPDATE. This could be negative after submitting negative
171 value to WINDOW_UPDATE */
172 int32_t recv_window_size;
173 /* The number of bytes consumed by the application and now is
174 subject to WINDOW_UPDATE. This is only used when auto
175 WINDOW_UPDATE is turned off. */
176 int32_t consumed_size;
177 /* The amount of recv_window_size cut using submitting negative
178 value to WINDOW_UPDATE */
179 int32_t recv_reduction;
180 /* window size for local flow control. It is initially set to
181 NGHTTP2_INITIAL_WINDOW_SIZE and could be increased/decreased by
182 submitting WINDOW_UPDATE. See nghttp2_submit_window_update(). */
183 int32_t local_window_size;
184 /* This is unpaid penalty (offset) when calculating cycle. */
185 uint32_t pending_penalty;
186 /* status code from remote server */
187 int16_t status_code;
188 /* Bitwise OR of zero or more nghttp2_http_flag values */
189 uint32_t http_flags;
190 /* This is bitwise-OR of 0 or more of nghttp2_stream_flag. */
191 uint8_t flags;
192 /* Bitwise OR of zero or more nghttp2_shut_flag values */
193 uint8_t shut_flags;
194 /* Nonzero if this stream has been queued to stream pointed by
195 dep_prev. We maintain the invariant that if a stream is queued,
196 then its ancestors, except for root, are also queued. This
197 invariant may break in fatal error condition. */
198 uint8_t queued;
199 /* This flag is used to reduce excessive queuing of WINDOW_UPDATE to
200 this stream. The nonzero does not necessarily mean WINDOW_UPDATE
201 is not queued. */
202 uint8_t window_update_queued;
203 /* extpri is a stream priority produced by nghttp2_extpri_to_uint8
204 used by RFC 9218 extensible priorities. */
205 uint8_t extpri;
206 /* http_extpri is a stream priority received in HTTP request header
207 fields and produced by nghttp2_extpri_to_uint8. */
208 uint8_t http_extpri;
209};
