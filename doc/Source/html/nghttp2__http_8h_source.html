36/*
37 * This function is called when HTTP header field |nv| in |frame| is
38 * received for |stream|. This function will validate |nv| against
39 * the current state of stream.
40 *
41 * This function returns 0 if it succeeds, or one of the following
42 * negative error codes:
43 *
44 * NGHTTP2_ERR_HTTP_HEADER
45 * Invalid HTTP header field was received.
46 * NGHTTP2_ERR_IGN_HTTP_HEADER
47 * Invalid HTTP header field was received but it can be treated as
48 * if it was not received because of compatibility reasons.
49 */
50int nghttp2_http_on_header(nghttp2_session *session, nghttp2_stream *stream,
51 nghttp2_frame *frame, nghttp2_hd_nv *nv,
52 int trailer);
